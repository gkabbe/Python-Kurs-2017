liste = []
a = 0
for i in range(11):
    liste.append(a)
    a += 2

# Methode 2

liste = []

for i in range(0, 21):
    if i % 2 == 0:
        liste.append(i)

print(liste)

    ergebnis = 0
    for arg in args:
        ergebnis += arg
    return ergebnis
